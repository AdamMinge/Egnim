# -------------------------------------------------------------------------------------------------- #
# -------------------------------------- Set CMake Version ----------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
cmake_minimum_required(VERSION 3.11)
# -------------------------------------------------------------------------------------------------- #
# -------------------------------------- Set files variable ---------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
set(source_root ${EGNIM_SOURCE_DIR}/src/egnim/server)
set(include_root ${EGNIM_SOURCE_DIR}/include/egnim/server)
set(resources_root ${EGNIM_SOURCE_DIR}/resources)

set(sources
        ${source_root}/main.cpp)

set(includes)

set(resources)
# -------------------------------------------------------------------------------------------------- #
# ----------------------------------- Find external libraries -------------------------------------- #
# -------------------------------------------------------------------------------------------------- #

# -------------------------------------------------------------------------------------------------- #
# ---------------------------------- create application target ------------------------------------- #
# -------------------------------------------------------------------------------------------------- #
egnim_add_application(egnim-server
        SOURCES ${sources} ${includes} ${resources}
        DEPENDS egnim::egnim-engine)

set_target_properties(egnim-server
        PROPERTIES CXX_STANDARD 20)
